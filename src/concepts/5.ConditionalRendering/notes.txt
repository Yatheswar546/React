Conditional Rendering:

=> To display components based on specific conditions.
=> The code is written similar way to ternary operator.

Eg:
1) Redirecting to Login/Signup Pages 
2) Read More buttons
3) Nav buttons 
4) Toggle effects (to show and hide a particular component)

Eg: Let us consider a scenario where I need to show & hide the counter 
    by clicking on counter button

Example Code:

// In ConditionalRendering.jsx

import { useState } from 'react';
import Counter from '../4.Props/Counter'

export default function ToggleCounter() {
    const [showCounter, setShowCounter] = useState(false);

    function handleToggle() {

        // 1st Method:
        // if(showCounter){
        //     setShowCounter(false);
        // } else{
        //     setShowCounter(true);
        // }
        
        // 2nd Method:
        // showCounter ? setShowCounter(false) : setShowCounter(true);
        
        // 3rd Method:
        setShowCounter(!showCounter)
    }
    return (
        <div>
            <button onClick={handleToggle}>
                {showCounter ? "Hide Counter" : "Show Counter"}
            </button>

            {showCounter && <Counter diff={1} name="Yathe" />}
        </div>
    )

}
