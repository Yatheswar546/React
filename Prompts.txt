Custom Hooks in React:

=> In React, a 'custom hook' is a reusable function that starts with the prefix "use" and 
    allows you to share and reuse stateful or side-effect
    ful logic across multiple components.
=> Custom hooks allow you to extract and share logic such as state management, side effects,
    and event handling without duplicating code.

    Custom Hooks Examples in React: https://github.com/streamich/react-use

NOTE: We cannot use custom Hook in a normal function we must use inside a Custom Hook(prefix "use") function only

useLocalStorageInput.jsx (inside hooks folder)

import { useState, useEffect } from 'react';

export const useLocalStorageInput = (key, defaultValue) => {

    const [input, setInput] = useState(localStorage.getItem(key) ?? defaultValue);

    useEffect(() => {
        localStorage.setItem(key, input);
    }, [input]);

    return {input, setInput} ;

};

Example 2 (Custom Hook that returns "window size"):

import { useState, useEffect } from 'react';

const useWindowSize = () => {
    const [windowsize, setWindowSize] = useState({
        width: window.innerWidth,
        height: window.innerHeight
    });

    useEffect(() => {
        const handleResize = () => {
            setWindowSize({
                width: window.innerWidth,
                height: window.innerHeight
            });
        };

        window.addEventListener("resize", handleResize);

        return () => {
            window.removeEventListener("resize", handleResize);
        };
    }, []);;

    return windowsize;
};

export default useWindowSize;

Example 3 (useLocalStorage custom Hook):

import { useState, useEffect } from 'react';

const useLocalStorage = (key, initialValue = "") => {

    const [value, setValue] = useState(() => {
        const storedValue = localStorage.getItem(key);
        
        return storedValue ? JSON.parse(storedValue) : initialValue;
    });

    useEffect(() => {
        localStorage.setItem(key, JSON.stringify(value));
    }, [key, value]);

    return [value, setValue];
};

export default useLocalStorage;
