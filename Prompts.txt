APIs:
1) Mock APIs (Mock Data):
    => Dummy Data
    => Used in project developing phase as backend team working on fetching the data.

Steps to create Mock APIs.
1) Create a data.js file with dummy data. (array of objects)
2) Create a component required to display your complete data. (eg: videos.jsx, blogs.jsx etc...)
3) Create a single component (VideoCard.jsx, BlogCard.jsx etc....)

    )    
}

In VideoCard.jsx file:
import Card from '@mui/material/Card';
import CardActions from '@mui/material/CardActions';
import CardContent from '@mui/material/CardContent';
import CardMedia from '@mui/material/CardMedia';
import Button from '@mui/material/Button';
import Typography from '@mui/material/Typography';

export default function VideoCard(data) {
    let {title, previewImage, videoLink, viewCount, releaseDate} = data;
    
    const navigate = useNavigate()

    const handleClick=()=>{
        window.location.href = videoLink;
    }

  return (
    <Card sx={{ maxWidth: 345 }}>
      <CardMedia
        sx={{ height: 140 }}
        image="{previewImage}"
        title="{title}"
      />
      <CardContent>
        <Typography gutterBottom variant="h5" component="div">
          {title}
        </Typography>
        <Typography variant="body2" sx={{ color: 'text.secondary' }}>
          Releasing on: {releaseDate}
        </Typography>
      </CardContent>
      <CardActions>
        <Button size="small">Views: {viewCount}</Button>
        <Button size="small" onClick={handleClick}>Watch Video</Button>
      </CardActions>
    </Card>
  );
}

In Videos.jsx file:

import React, { useEffect } from "react";
import VideoCard from "./VideoCard";
import { Grid } from "@mui/material";
import axios from "axios";
import { enqueueSnackbar } from "notistack";

export default function Videos() {

    const [videos, setVideos] = useState([])

    async function fetchVideos(){
        try{
            let res = await axios.get("https://............");
            setVideos(res.data.videos);
            enqueueSnackbar("Fetched videos", {
                variant: "success", 
                preventDuplicate: true,
                anchorOrigin: {horizontal: "right", vertical: "top"}
            });
        }
        catch(err){
            console.log(err, "err");
            enqueueSnackbar(err.response.data.message, {variant: "error"});
        }
    }

    useEffect(() => {
        fetchVideos();
    }, []);

    return(
        <Grid container spacing={2} sx={{ justifyContent: "center" }}>
            {videos.map(video) => (
                <VideoCard data={video} key={video.id}/>
            )}
        </Grid>
    )    
}
